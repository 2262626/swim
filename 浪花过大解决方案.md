核心问题本质
游泳时水花 / 波纹会让 MediaPipe 出现 3 种故障：
关键点丢失（腿 / 手看不见）
关键点漂移（识别到水花上）
整帧失效（没有 pose 结果）
一、前端立刻能用的 5 个解决方案（优先级从高到低）
方案 1：时域平滑 + 历史帧插值（最强、最推荐）
原理
人游泳动作是连续运动，不会瞬间瞬移。当前一帧有水花、这帧识别不到 → 用前几帧的位置预测 / 平滑过渡。
前端可直接实现
维护一个 关键点队列（buffer），存最近 5~10 帧
当某帧关键点 visibility < 0.5（被水花挡住）
自动使用历史帧的加权平均代替
相当于 “补帧”“抗抖动”
核心代码（直接复制）
javascript
运行
// 缓存最近 8 帧关键点
let poseBuffer = [];
const BUFFER_SIZE = 8;

pose.onResults((results) => {
  const keypoints = results.poseLandmarks;

  // 1. 推入缓存
  poseBuffer.push(keypoints);
  if (poseBuffer.length > BUFFER_SIZE) poseBuffer.shift();

  // 2. 对每个关键点做平滑 + 遮挡修复
  const smoothedPoints = smoothPosePoints(poseBuffer);
  
  // 3. 使用 smoothedPoints 做判断，不再用原始点
  analyzeButterfly(smoothedPoints);
});

// 平滑 + 遮挡修复函数
function smoothPosePoints(buffer) {
  const result = [];
  const pointCount = 33;

  for (let i = 0; i < pointCount; i++) {
    let sumX = 0, sumY = 0, count = 0;

    // 只取可见度高的帧
    buffer.forEach(frame => {
      const p = frame[i];
      if (p?.visibility > 0.5) {
        sumX += p.x;
        sumY += p.y;
        count++;
      }
    });

    if (count === 0) {
      // 完全被遮挡 → 保留最后一帧有效位置
      return getLastValidPoint(buffer, i);
    }

    result[i] = {
      x: sumX / count,
      y: sumY / count,
      visibility: 1.0
    };
  }
  return result;
}

// 画布预处理：降低水花干扰
function preprocessCanvas(ctx, width, height) {
  const imageData = ctx.getImageData(0, 0, width, height);
  const data = imageData.data;

  for (let i = 0; i < data.length; i += 4) {
    // 降低高光（水花）
    if (data[i] > 200 && data[i+1] > 200 && data[i+2] > 200) {
      data[i] *= 0.7;    // R
      data[i+1] *= 0.7;  // G
      data[i+2] *= 0.7;  // B
    }
  }
  ctx.putImageData(imageData, 0, 0);
}
